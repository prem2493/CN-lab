#include <stdio.h>
#include <stdlib.h>
#include <math.h>

// Function to calculate the greatest common divisor
int gcd(int a, int b) {
    while (b != 0) {
        int temp = b;
        b = a % b;
        a = temp;
    }
    return a;
}

// Function to calculate (base^exponent) % modulus
long long mod_exp(long long base, long long exponent, long long modulus) {
    long long result = 1;
    while (exponent > 0) {
        if (exponent % 2 == 1) {
            result = (result * base) % modulus;
        }
        base = (base * base) % modulus;
        exponent = exponent / 2;
    }
    return result;
}

int main() {
    long long p, q, n, phi, e, d;
    long long encrypted, decrypted;
    long long message;

    // Step 1: Ask user for two large prime numbers
    printf("Enter a large prime number p: ");
    scanf("%lld", &p);
    printf("Enter another large prime number q: ");
    scanf("%lld", &q);

    // Step 2: Calculate n = p * q
    n = p * q;

    // Step 3: Calculate the Euler's Totient Function, phi(n)
    phi = (p - 1) * (q - 1);

    // Step 4: Dynamically choose e such that 1 < e < phi and gcd(e, phi) = 1
    e = 2;
    while (gcd(e, phi) != 1) {
        e++;
    }

    printf("Chosen e = %lld\n", e);

    // Step 5: Compute d, the modular multiplicative inverse of e modulo phi
    // Using a brute force approach to find d such that (d * e) % phi = 1
    d = 0;
    while ((e * d) % phi != 1) {
        d++;
    }

    // Display public and private keys
    printf("Public Key (n = %lld, e = %lld)\n", n, e);
    printf("Private Key (n = %lld, d = %lld)\n", n, d);

    //Step 6: Get the message to encrypt (as an integer)
    do {
        printf("Enter a message to encrypt (as an integer, < n = %d): ", n);
        scanf("%d", &message);
    } while (message >= n);  // Ensure the message is smaller than n

    // Step 7: Encrypt the message using: encrypted = (message^e) % n
    encrypted = mod_exp(message, e, n);
    printf("Encrypted message: %lld\n", encrypted);

    // Step 8: Decrypt the message using: decrypted = (encrypted^d) % n
    decrypted = mod_exp(encrypted, d, n);
    printf("Decrypted message: %lld\n", decrypted);

    return 0;
}
